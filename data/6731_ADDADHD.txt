<!-- /wp:heading -->

<!-- wp:list -->
<ul><li>Inattentiveness</li><li>Hyperactivity</li><li>Impulsiveness</li><li>Short attention span</li><li>Careless</li><li>Forgetful</li><li>Restlessness</li><li>unable to listen or carry out instructions</li><li>Difficulty in organizing tasks</li></ul>
